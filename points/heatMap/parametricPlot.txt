abarf(n,b,beta) = b*(N-1)*(N-1)/(N-n)/(N-1+n*beta)
nstarf(abar,b) = N - b/aa*(N-1)
betaf(n,b,abar) = ( b*(N-1)/abar/(N-n) - 1.0 )*(N-1)/n
bf(n,abar,beta) = abar * (N-n) * ( N-1 + n*beta ) / (N-1.0) / (N-1.0)
GNUTERM = "wxt"
N = 100.0
GPFUN_abarf = "abarf(n,b,beta) = b*(N-1)*(N-1)/(N-n)/(N-1+n*beta)"
GPFUN_nstarf = "nstarf(abar,b) = N - b/aa*(N-1)"
GPFUN_betaf = "betaf(n,b,abar) = ( b*(N-1)/abar/(N-n) - 1.0 )*(N-1)/n"
GPFUN_bf = "bf(n,abar,beta) = abar * (N-n) * ( N-1 + n*beta ) / (N-1.0) / (N-1.0)"
beta0 = 1.0
bb = 0.8
plot [-(N-2):N-2][0:1][-5:105] 	abarf(t,bb,10),t lw 3 title '{/Symbol b}=10'
